#!/usr/bin/env node
/**
 * extractEfdToBin.js
 *
 * Extracts the raw payload from an EFD file (the same output
 * that the original Windows EFD_Reader.exe produces) and writes it
 * as a .bin file.
 *
 * Usage:
 *   node extractEfdToBin.js <input.efd> [output.bin]
 *
 * Example:
 *   node extractEfdToBin.js 68341609ad.efd
 *   → creates 68341609ad.bin
 */

const fs = require('fs');
const path = require('path');

// -------------------------------------------------------------------
// Simple EBML‑style size decoder (used by the EFD format)
// The size field is variable‑length; the high‑bit of each byte indicates
// whether more bytes follow. The remaining 7 bits contribute to the size.
function readEbmlSize(buf, offset) {
  let length = 0;
  let size = 0;
  while (offset + length < buf.length) {
    const byte = buf[offset + length];
    size = (size << 7) | (byte & 0x7f);
    length++;
    if ((byte & 0x80) === 0) break; // MSB cleared => last size byte
  }
  return { size, length };
}

// -------------------------------------------------------------------
// Locate the payload element (ID 0x42F5) inside the file.
// Returns {offset, size} where `offset` points to the first payload byte.
function locatePayload(buf) {
  const PAYLOAD_ID = Buffer.from([0x42, 0xF5]); // two‑byte ID used in our samples

  let pos = 0;
  while (pos < buf.length - 1) {
    if (buf[pos] === PAYLOAD_ID[0] && buf[pos + 1] === PAYLOAD_ID[1]) {
      // Found the element ID – now read its EBML size.
      const sizeInfo = readEbmlSize(buf, pos + 2);
      const payloadOffset = pos + 2 + sizeInfo.length;
      return { offset: payloadOffset, size: sizeInfo.size };
    }
    pos++;
  }
  throw new Error('Payload element (0x42F5) not found in the EFD file');
}

// -------------------------------------------------------------------
function main() {
  const args = process.argv.slice(2);
  if (args.length < 1) {
    console.error('Usage: node extractEfdToBin.js <input.efd> [output.bin]');
    process.exit(1);
  }

  const inputPath = args[0];
  const outputPath =
    args[1] || path.basename(inputPath, '.efd') + '.bin';

  const efdData = fs.readFileSync(inputPath);
  const { offset, size } = locatePayload(efdData);

  console.log(
    `Payload located at offset 0x${offset.toString(
      16
    )} (size ${size} bytes)`
  );

  const payload = efdData.slice(offset, offset + size);
  fs.writeFileSync(outputPath, payload);
  console.log(`Wrote binary payload to ${outputPath}`);
}

main();
How it works

Searches for the two‑byte EBML element ID 0x42F5 – this is the “Payload” element used in the EFD files (confirmed by the existing SRT‑Lab parser).
Decodes the following variable‑length size field using standard EBML rules (high‑bit continuation).
Slices the exact payload bytes (buf.slice(offset, offset + size)).
Writes those bytes unmodified to <original‑name>.bin.
No decryption key is required; the output is precisely the encrypted payload that EFD_Reader.exe writes to disk.
